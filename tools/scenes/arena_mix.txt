# Hybrid scene combining spheres and a mesh obstacle
load Arena assets\cube.obj
move Arena 0.0 -0.5 0.0
sphere BallA 0.3 -1.5 0.0 0.0
sphere BallB 0.3  1.5 0.0 0.0
collide BallA Arena
collide BallB Arena
simulate BallA Arena 0.12 0.0 0.0 15 60
simulate BallB Arena -0.12 0.0 0.0 15 60
ccd BallA Arena 0.5 0.0 0.0
